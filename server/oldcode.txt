void getIpAddressString(char* pIpAddress, char* pIpVersion, struct sockaddr_storage* pTheirAddress) {
    void *addr;
    if (pTheirAddress->ss_family == AF_INET) {
        struct sockaddr_in *ipv4 = (struct sockaddr_in *)pTheirAddress;
        addr = &(ipv4->sin_addr);
        strcpy(pIpVersion, "IPv4");
    } else {
        struct sockaddr_in6 *ipv6 = (struct sockaddr_in6 *)pTheirAddress;
        addr = &(ipv6->sin6_addr);
        strcpy(pIpVersion, "IPv6");
    }
    inet_ntop(pTheirAddress->ss_family, addr, pIpAddress, INET6_ADDRSTRLEN);
}
